SINGLE INHERTITANCE:

class grandfather {
    void oldbike () {
        System.out.println("Splender");
    }
}
class father extends grandfather {  
    void bike () {
        System.out.println("Discover");
    }
}
public class main {
    public static void main(String[] args) {
        father obj = new father();
        obj.bike();
        obj.oldbike ();
    }
}

MULTILEVEL INHERITANCE:

public class inheritence {
	public static void main(String[] args){
		son1 s1 = new son1();
		s1.bike();
		son2 s2 = new son2();
		s2.oldbike();
		s2.bike();
	}
}
